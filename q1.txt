billing_address_city, shipping_address_city	(select trim(category_lang.localetext) from crmuser.category_lang,crmuser.categories where category_lang.CATEGORYID=categories.CATEGORYID and categories.CATEGORYTYPE='CITY' and  categories.VALUE=(select  city from crmuser.address where orgkey=acctcustids(k) and ADDRESSCATEGORY='Mailing' and (end_date is null or (end_date>='31-12-2099')) and rownum=1))
billing_address_state, shipping_address_state	(select category_lang.localetext from crmuser.category_lang,crmuser.categories where category_lang.CATEGORYID=categories.CATEGORYID and categories.CATEGORYTYPE='STATE' and categories.VALUE=(select state from crmuser.address where orgkey=acctcustids(k) and ADDRESSCATEGORY='Mailing' and (end_date is null or (end_date>='31-12-2099')) and rownum=1))
billing_address_country, shipping_address_country	(select category_lang.localetext from crmuser.category_lang,crmuser.categories where category_lang.CATEGORYID=categories.CATEGORYID and categories.CATEGORYTYPE='COUNTRY' and  categories.VALUE=(select country from crmuser.address where orgkey=acctcustids(k) and ADDRESSCATEGORY='Mailing' and (end_date is null or (end_date>='31-12-2099')) and rownum=1))
phone_office	(Select PHONENO from crmuser.phoneemail p where p.orgkey = a.CUSTT and PHONEOREMAIL = 'PHONE' and PHONEEMAILTYPE = 'HOMEPH1' and rownum < 2)
phone_alternate	(select PHONENO from crmuser.phoneemail p where p.orgkey = a.CUSTT and PHONEOREMAIL = 'PHONE' and PHONEEMAILTYPE = 'WORKPH1' and rownum < 2)
